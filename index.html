Not Using Map


int[] myArray = [4, 5, 6];
int[] newArray = new Array[];

// gotta add 5 to each element of the array
for (int i = 0, i < myArray.length, i++) {
  newArray[i] = myArray[i] + 5;
}

print(newArray);
// [9, 10, 11]

Map


const myArray = [4, 5, 6];
const newArray = _.map(myArray, i => i + 5);

print(newArray);
// [9, 10, 11]

The Versatility!


_.map("hello", _.identity)

Reduce


_.some([2, 5, 8], i => i > 5)
// true

Compact


_.compact(["Hello", null, 1, ""])
// ["Hello", 1]

Intersection


_.intersection([1, 4, 7], [7, 8, 9])
// [7]

Kebab-Case

// "i-love-java-script"

Kebab-Case


_.some(
  _.kebabCase("I love JavaScript").split("-"),
  word => word.length > 8
